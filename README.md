        // Initialize AOS animations
        document.addEventListener('DOMContentLoaded', function() {

                duration: 800,
                easing: 'ease-in-out',
                once: true
            });

            // Investment Calculator
            const calculateBtn = document.getElementById('calculate-btn');
            const resultDiv = document.getElementById('result');
            
            calculateBtn.addEventListener('click', function() {
                const investmentAmount = parseFloat(document.getElementById('investment-amount').value);
                const fundType = document.getElementById('fund-type').value;
                
                if (isNaN(investmentAmount) || investmentAmount <= 0) {
                    resultDiv.innerHTML = 'Please enter a valid investment amount.';
                    resultDiv.classList.remove('hidden');
                    return;
                }
                
                let days, dailyRate, totalEarnings, totalValue;
                
                if (fundType === 'stanlib') {
                    days = 360;
                    dailyRate = 0.0459;
                } else {
                    days = 7;
                    dailyRate = 0.0057;
                }
                
                totalEarnings = investmentAmount * dailyRate * days;
                totalValue = investmentAmount + totalEarnings;
                
                resultDiv.innerHTML = `
                    <div class="p-4 bg-gray-100 rounded-lg">
                        <p>Investment Amount: <span class="text-primary">R ${investmentAmount.toLocaleString()}</span></p>
                        <p>Investment Period: <span class="text-primary">${days} days</span></p>
                        <p>Total Earnings: <span class="text-green-600 font-bold">R ${totalEarnings.toLocaleString()}</span></p>
                        <p>Total Value (Principal + Earnings): <span class="text-primary font-bold">R ${totalValue.toLocaleString()}</span></p>
                    </div>
                `;
                resultDiv.classList.remove('hidden');
            });

            // Smooth scrolling for navigation links
            document.querySelectorAll('a[href^="#"]').forEach(anchor => {
